-- 1. CLEANUP (Optional: Drops tables if they exist to allow for clean re-runs)
DROP TABLE IF EXISTS CLIENT_COURSE;
DROP TABLE IF EXISTS APPLICATION;
DROP TABLE IF EXISTS CLIENT_SERVICE;
DROP TABLE IF EXISTS CRIMINAL_RECORD;
DROP TABLE IF EXISTS JOB_OPENING;
DROP TABLE IF EXISTS SERVICE;
DROP TABLE IF EXISTS COURSE;
DROP TABLE IF EXISTS EMPLOYER;
DROP TABLE IF EXISTS CLIENT;

-- 2. CREATE TABLES (Ordered by dependency)
CREATE TABLE CLIENT (
   Client_ID     INT           PRIMARY KEY,
   L_Name        NVARCHAR(35),
   F_Name        NVARCHAR(35),
   Gender        CHAR(1)        CHECK(Gender IN ('M','F','O')),
   DOB           DATE,
   Date_joined   DATE,
   Mentor_ID_FK  INT            NULL,
   CONSTRAINT FK_CLIENT_MENTOR FOREIGN KEY (Mentor_ID_FK) REFERENCES CLIENT(Client_ID)
);

CREATE TABLE CRIMINAL_RECORD (
   Crime_ID            INT            PRIMARY KEY,
   Type_of_crime       NVARCHAR(35),
   Date_commited       DATE,
   Date_sentenced      DATE,
   Length_of_sentence  DECIMAL(5,2),
   Client_ID_FK        INT,
   CONSTRAINT FK_CRIMINAL_CLIENT FOREIGN KEY (Client_ID_FK) REFERENCES CLIENT(Client_ID)
);

CREATE TABLE EMPLOYER (
   Employer_ID   INT           PRIMARY KEY,
   Company_name  NVARCHAR(35),
   Contact_person NVARCHAR(35),
   Phone_number  NVARCHAR(15)
);

CREATE TABLE JOB_OPENING (
   Opening_ID     INT           PRIMARY KEY,
   Title          NVARCHAR(35),
   Pay_rate       DECIMAL(5,2),
   Employer_ID_FK INT,
   CONSTRAINT FK_JOB_EMPLOYER FOREIGN KEY (Employer_ID_FK) REFERENCES EMPLOYER(Employer_ID)
);

CREATE TABLE SERVICE (
   Service_ID    INT           PRIMARY KEY,
   Service_name  NVARCHAR(35),
   Provider      NVARCHAR(35)
);

CREATE TABLE CLIENT_SERVICE (
   Client_ID_FK  INT,
   Service_ID_FK INT,
   Date_started  DATE,
   CONSTRAINT PK_CLIENT_SERVICE PRIMARY KEY (Client_ID_FK, Service_ID_FK),
   CONSTRAINT FK_CS_CLIENT FOREIGN KEY (Client_ID_FK) REFERENCES CLIENT(Client_ID),
   CONSTRAINT FK_CS_SERVICE FOREIGN KEY (Service_ID_FK) REFERENCES SERVICE(Service_ID)
);

CREATE TABLE APPLICATION (
   Client_ID_FK    INT,
   Opening_ID_FK   INT,
   Date_applied    DATE,
   Successful      BIT,
   CONSTRAINT PK_APPLICATION PRIMARY KEY (Client_ID_FK, Opening_ID_FK),
   CONSTRAINT FK_APP_CLIENT FOREIGN KEY (Client_ID_FK) REFERENCES CLIENT(Client_ID),
   CONSTRAINT FK_APP_JOB FOREIGN KEY (Opening_ID_FK) REFERENCES JOB_OPENING(Opening_ID)
);
